## Explicacion de primer problema.

* [A. Determine Line](https://codeforces.com/contest/1056/problem/A)

```cpp
#include <bits/stdc++.h>
using namespace std;
int n;
int cnt[102];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		int m;
		scanf("%d", &m);
		for (int i = 0; i < m; ++i) {
			int x;
			scanf("%d", &x);
			cnt[x] += 1;
		}
	}
	for (int i = 1; i <= 100; ++i) {
		if (cnt[i] == n) {
			cout << i << " ";
		}
	} 
	cout << endl;
	return 0;
}
```
